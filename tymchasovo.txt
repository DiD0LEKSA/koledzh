package org.oleksijson.core;

import java.io.BufferedReader;
import java.io.File;
import java.io.FileReader;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;

public class IntParser {

    public List<LocalizationEntry> parse(File file) throws IOException {
        List<LocalizationEntry> entries = new ArrayList<>();

        try (BufferedReader br = new BufferedReader(new FileReader(file))) {
            String line;
            while ((line = br.readLine()) != null) {
                line = line.trim();

                // Пропускаємо порожні рядки, коментарі та назви секцій [Секція]
                if (line.isEmpty() || line.startsWith(";") || line.startsWith("[")) {
                    continue;
                }

                // Шукаємо знак рівності
                if (line.contains("=")) {
                    int eqIndex = line.indexOf("=");
                    String key = line.substring(0, eqIndex).trim();
                    String value = line.substring(eqIndex + 1).trim();

                    // Прибираємо лапки, якщо вони є навколо тексту
                    if (value.startsWith("\"") && value.endsWith("\"")) {
                        value = value.substring(1, value.length() - 1);
                    }

                    entries.add(new LocalizationEntry(key, value));
                }
            }
        }
        return entries;
    }
}

package org.oleksijson.core.model;

/**
 * Модель однієї одиниці локалізації.
 * key - назва рядка в коді гри (наприклад, "Main_Menu_Title")
 * value - сам текст (наприклад, "Головне меню")
 */
public class LocalizationEntry {
    private final String key;
    private final String value;

    public LocalizationEntry(String key, String value) {
        this.key = key;
        this.value = value;
    }

    // Геттери (вони потрібні для Jackson, щоб він міг зробити JSON)
    public String getKey() { return key; }
    public String getValue() { return value; }

    @Override
    public String toString() {
        return "Key: " + key + " | Value: " + value;
    }
}

package org.oleksijson.core.exporter.json;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.SerializationFeature;
import org.oleksijson.core.model.LocalizationEntry;

import java.io.File;
import java.io.IOException;
import java.util.List;

public class JsonExporter {
    private final ObjectMapper mapper;

    public JsonExporter() {
        this.mapper = new ObjectMapper();
        // Робимо JSON гарним (з відступами), а не в один рядок
        this.mapper.enable(SerializationFeature.INDENT_OUTPUT);
    }

    public void export(List<LocalizationEntry> data, File outputFile) throws IOException {
        mapper.writeValue(outputFile, data);
    }
}

package org.oleksijson.ui;

import javafx.fxml.FXML;
import javafx.scene.control.*;
import javafx.stage.DirectoryChooser;
import javafx.stage.FileChooser;
import javafx.stage.Stage;
import org.oleksijson.core.service.ConversionService;
import org.oleksijson.utils.AlertHelper;
import javafx.scene.control.CheckMenuItem;

import java.io.File;
import java.io.IOException;

public class MainController {

    @FXML private TextField pathField;
    @FXML private Label statusLabel;
    @FXML private CheckMenuItem deleteOriginalsMenu;
    @FXML private Button convertToJsonBtn;
    @FXML private Button convertToIntBtn;

    private final ConversionService conversionService = new ConversionService();
    private File selectedPath;

    @FXML
    private void initialize() {
        // Початковий стан кнопок — вимкнено, поки нічого не обрано
        updateButtonStates(true);
    }

    @FXML
    private void onChooseFile() {
        FileChooser fileChooser = new FileChooser();
        fileChooser.setTitle("Оберіть файл локалізації");
        // Додаємо фільтри, щоб не бачити зайвого
        fileChooser.getExtensionFilters().addAll(
                new FileChooser.ExtensionFilter("Localization Files", "*.int", "*.json", "*.ukr", "*.ger")
        );

        File file = fileChooser.showOpenDialog(new Stage());
        if (file != null) {
            processPathSelection(file);
        }
    }

    @FXML
    private void onChooseFolder() {
        DirectoryChooser directoryChooser = new DirectoryChooser();
        directoryChooser.setTitle("Оберіть папку з файлами");
        File folder = directoryChooser.showDialog(new Stage());
        if (folder != null) {
            processPathSelection(folder);
        }
    }

    private void processPathSelection(File file) {
        selectedPath = file;
        pathField.setText(selectedPath.getAbsolutePath());
        statusLabel.setText("Об’єкт обрано: " + selectedPath.getName());
        updateButtonStates(false);
    }

    @FXML
    private void onConvertToJson() {
        runTask(() -> {
            if (selectedPath.isDirectory()) {
                conversionService.convertFolderToJson(selectedPath, deleteOriginalsCheck.isSelected());
            } else {
                File outputFile = new File(selectedPath.getParent(), selectedPath.getName().replaceAll("(?i)\\.(int|ukr|ger)$", ".json"));
                conversionService.convertIntToJson(selectedPath, outputFile);
            }
        }, "Конвертація в JSON успішна!");
    }

    @FXML
    private void onConvertToInt() {
        runTask(() -> {
            if (selectedPath.isDirectory()) {
                conversionService.convertFolderToInt(selectedPath, deleteOriginalsCheck.isSelected());
            } else {
                File outputFile = new File(selectedPath.getParent(), selectedPath.getName().replaceAll("(?i)\\.json$", ".int"));
                conversionService.convertJsonToInt(selectedPath, outputFile);
            }
        }, "Конвертація в INT успішна!");
    }

    private void runTask(RunnableWithException task, String successMessage) {
        try {
            task.run();
            statusLabel.setText("Готово!");
            AlertHelper.showInfo("Успіх", successMessage);
        } catch (IOException e) {
            statusLabel.setText("Помилка!");
            AlertHelper.showError("Помилка", "Сталася проблема при обробці: " + e.getMessage());
        }
    }

    private void updateButtonStates(boolean disabled) {
        convertToJsonBtn.setDisable(disabled);
        convertToIntBtn.setDisable(disabled);
    }

    @FunctionalInterface
    private interface RunnableWithException {
        void run() throws IOException;
    }
}

package org.oleksijson.core.model;

import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
import com.fasterxml.jackson.annotation.JsonPropertyOrder;

/**
 * Оновлена модель для OleksiJson.
 * Порядок полів: section -> key -> source -> translation.
 */
@JsonPropertyOrder({ "section", "key", "source", "translation" })
@JsonIgnoreProperties(ignoreUnknown = true) // Ігноруй все, чого немає в цьому класі
public class LocalizationEntry {
    private String section;
    private String key;
    private String source;
    private String translation;

    // Порожній конструктор (завжди потрібен для Jackson)
    public LocalizationEntry() {}

    // Конструктор для створення записів під час парсингу .int
    public LocalizationEntry(String section, String key, String source, String translation) {
        this.section = section;
        this.key = key;
        this.source = source;
        this.translation = translation;
    }

    // Геттери та сеттери
    public String getSection() { return section; }
    public void setSection(String section) { this.section = section; }

    public String getKey() { return key; }
    public void setKey(String key) { this.key = key; }

    public String getSource() { return source; }
    public void setSource(String source) { this.source = source; }

    public String getTranslation() { return translation; }
    public void setTranslation(String translation) { this.translation = translation; }
}

package org.oleksijson.core.service;

import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;
import java.io.InputStream;

public class LocalizationManager {
    private static JsonNode translations;
    private static final ObjectMapper mapper = new ObjectMapper();

    public static void load(String lang) {
        // 1. Формуємо шлях. ВАЖЛИВО: без "/" на початку для getClassLoader()
        String path = "localizations/" + lang + ".json";

        System.out.println("🔍 Шукаю локалізацію: " + path);

        // 2. Спроба завантажити через ClassLoader (найкращий варіант для Maven)
        InputStream is = LocalizationManager.class.getClassLoader().getResourceAsStream(path);

        // 3. Якщо не знайшло, пробуємо через сам клас (із косою рискою)
        if (is == null) {
            is = LocalizationManager.class.getResourceAsStream("/" + path);
        }

        if (is != null) {
            try {
                translations = mapper.readTree(is);
                System.out.println("✅ Файл знайдено і завантажено!");
            } catch (Exception e) {
                System.err.println("❌ Помилка синтаксису JSON: " + e.getMessage());
            }
        } else {
            System.err.println("❌ КРИТИЧНО: Файл не знайдено за шляхом: " + path);
            // Створюємо порожній вузол, щоб програма не падала з помилкою "in is null"
            translations = mapper.createObjectNode();
        }
    }

    public static String get(String path) {
        if (translations == null) return path;

        // Дозволяє діставати значення через крапку: "menu.settings"
        String[] keys = path.split("\\.");
        JsonNode node = translations;
        for (String key : keys) {
            node = node.get(key);
            if (node == null){
            System.err.println("❌ Ключ не знайдено: " + path);
            return path; // Повертаємо сам шлях, щоб не було порожнього місця в UI
                }
        }
        return node.asText();
    }
}

    private final IntParser intParser = new IntParser();
    private final JsonParser jsonParser = new JsonParser();
    private final IntExporter intExporter = new IntExporter();
    private final JsonExporter jsonExporter = new JsonExporter();

    /**
     * Конвертує .int (UE) файл у JSON
     */
    public void convertIntToJson(File inputFile, File outputFile) throws IOException {
        List<LocalizationEntry> entries = intParser.parse(inputFile);
        jsonExporter.export(entries, outputFile);
    }

    /**
     * Конвертує JSON файл назад у .int (UE)
     */
    public void convertJsonToInt(File inputFile, File outputFile) throws IOException {
        List<LocalizationEntry> entries = jsonParser.parse(inputFile);
        intExporter.export(entries, outputFile);
    }

    // --- МАСОВА КОНВЕРТАЦІЯ (ТВОЇ "ТЕКИ") ---

    public void convertFolderToJson(File folder, boolean deleteOriginals) throws IOException {
        File[] files = folder.listFiles((_, name) -> name.toLowerCase().endsWith(".int"));
        if (files == null) return;

        for (File file : files) {
            File outputFile = new File(file.getParent(), file.getName().replaceAll("(?i)\\.int$", ".json"));
            convertIntToJson(file, outputFile);

            if (deleteOriginals) {
                boolean deleted = file.delete(); // Перевіряємо видалення, щоб не було варнінгу
            }
        }
    }

    public void convertFolderToInt(File folder, boolean deleteOriginals) throws IOException {
        File[] files = folder.listFiles((_, name) -> name.toLowerCase().endsWith(".json"));
        if (files == null) return;

        for (File file : files) {
            File outputFile = new File(file.getParent(), file.getName().replaceAll("(?i)\\.json$", ".int"));
            convertJsonToInt(file, outputFile);

            if (deleteOriginals) {
                boolean deleted = file.delete();
            }
        }
    }
}
